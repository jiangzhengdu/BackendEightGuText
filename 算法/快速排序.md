快速排序
====

    static void quickSort(int[] array, int low ,int high){
            if(high<low) return;
            int i,j,index;
            i=low;
            j=high;
            index=array[i];
            int temp = 0;
            while(i<j){
                while(array[j]>=index&&i<j)
                    j--;
                while(array[i]<=index&&i<j)
                    i++;
                if(i<j) {
                    temp = array[j];
                    array[j] = array[i];
                    array[i] = temp;
                }
            }
            array[low]=array[i];
            array[i]=index;
            quickSort(array,low,i-1);
            quickSort(array,i+1,high);
    }
